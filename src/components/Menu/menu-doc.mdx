## Menu
根据需要展示不同的 `Menu`, 以满足不同系统的布局展示。

import { Story, Meta, ArgsTable, Source } from '@storybook/addon-docs/blocks';
import Menu, { MenuProps } from './menu';
import dedent from 'ts-dedent';

<Meta title="Components/Menu" component={Menu} />

### 使用

<Source
  language='jsx'
  code={
    dedent`
      <Menu mode={mode} 
        defaultOpenSubMenus={defaultOpenSubMenus} 
        defaultIndex={defaultIndex} 
        onSelect={(index) => {console.log(index)}} 
      >
        <MenuItem>
            菜单 1
        </MenuItem>
        <MenuItem>
            菜单 2
        </MenuItem>
        <MenuItem>
            菜单 3
        </MenuItem>
        <SubMenu title='菜单 4'>
            <MenuItem>
                子菜单项 1
            </MenuItem>
            <MenuItem>
                子菜单项 2
            </MenuItem>
            <MenuItem>
                子菜单项 3
            </MenuItem>
        </SubMenu>
      </Menu>`
  }
/>
